# Trait  

- a trait is a bag of vars & funcs with defaults 
- trait need to declare all vars & funcs it needs
- trait can access global variables in the scope where it is defined 
- resolve to a trait specific func using :: 
- class mediates conflicts between traits 


trait Fortifiable
    # private vars are not accessible by the class 
    # and do not cause conflicts  
    _id: int 
    health: int

    # trait cannot have initialize, only default
    fun default ()        
        me.health = 100
    
    fun run ()
        pass
    
    fun defend ()
        pass


trait Living
    health: int
    
    fun default()
        me.health = 50

    fun run ()
        pass


class Worker is Fortifiable, Living
    # Var Conflict detected: 'health' exists in two traits
    # MUST override in class
    health: int 


impl Worker 
    fun default()
        me.health = 75 


impl Worker is Fortifiable
    fun defend()
        pass 


impl Worker  
    # Fun Conflict detected: 'run' exists in two traits
    # MUST override in class
    fun run ()
        # You decide how to reconcile the logic
        me::Fortifiable.run()
        me::Living.run()
